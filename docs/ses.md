normalize_from_quantiles = function(X, vns=colnames(X), prob_limits=0.95 ) {

  for (j in 1:length(vns)) {
    x = X[,j]    
		pr = ecdf(x) ( x )
		
    pr_one_tail = (1 - prob_limits) / 2 
    prlb = pr_one_tail
    prub = 1 - pr_one_tail

    u = which( pr < prlb )
    if (length(u) > 0 ) pr[u] = prlb
    
    u = which( pr > prub )
    if (length(u) > 0 ) pr[u] = prub
    
    X[,i] = qnorm( pr, mean=0.0, sd=1.0 )
  } 
  return(X)
}
